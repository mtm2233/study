    class Solution {
    public int mySqrt(int x) {
    if (x == 0 || x == 1) {
    return x;
    }

    int start = 0;
    int end = (x >> 1) + 1;

    while (start + 1 < end) {
    final int mid = start + ((end - start) >> 1);
    final int quotient = x / mid;
    if (quotient == mid) {
    return mid;
    } else if (quotient < mid) {
    end = mid;
    } else {
    start = mid;
    }
    }

    return start;
    }
